24  package microsoft.exchange.webservices.data.autodiscover.exception;
25  
26  import microsoft.exchange.webservices.data.autodiscover.exception.error.AutodiscoverError;
27  import microsoft.exchange.webservices.data.core.exception.service.remote.ServiceRemoteException;
28  
29  /**
30   * Represents an exception that is thrown when the Autodiscover service returns
31   * an error.
32   */
33  public class AutodiscoverRemoteException extends ServiceRemoteException {
34  
35    /**
36     * Constant serialized ID used for compatibility.
37     */
38    private static final long serialVersionUID = 1L;
39  
40    /**
41     * The error.
42     */
43    private AutodiscoverError error;
44  
45    /**
46     * Initializes a new instance of the class.
47     *
48     * @param error the error
49     */
50    public AutodiscoverRemoteException(AutodiscoverError error) {
51      super();
52      this.error = error;
53    }
54  
55    /**
56     * Initializes a new instance of the class.
57     *
58     * @param message the message
59     * @param error   the error
60     */
61    public AutodiscoverRemoteException(String message, AutodiscoverError error) {
62      super(message);
63      this.error = error;
64    }
65  
66    /**
67     * Initializes a new instance of the class.
68     *
69     * @param message        the message
70     * @param error          the error
71     * @param innerException the inner exception
72     */
73    public AutodiscoverRemoteException(String message, AutodiscoverError error,
74        Exception innerException) {
75      super(message, innerException);
76      this.error = error;
77    }
78  
79    /**
80     * Gets the error.
81     *
82     * @return the error
83     */
84    public AutodiscoverError getError() {
85      return this.error;
86    }
87  }
