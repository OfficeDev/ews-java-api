586   /**
587    * Parses the XML using the specified reader and creates an Outlook
588    * protocol.
589    *
590    * @param reader The reader.
591    * @throws Exception the exception
592    */
593   protected void loadFromXml(EwsXmlReader reader)
594       throws Exception {
595     do {
596       reader.read();
597       if (reader.getNodeType().getNodeType() == XmlNodeType.START_ELEMENT) {
598         if (reader.getLocalName().equals(XmlElementNames.Type)) {
599           this.setProtocolType(OutlookProtocol.
600               protocolNameToType(reader.readElementValue()));
601         } else if (reader.getLocalName().equals(XmlElementNames.AuthPackage)) {
602           this.authPackage = reader.readElementValue();
603         } else if (reader.getLocalName().equals(XmlElementNames.Server)) {
604           this.server = reader.readElementValue();
605         } else if (reader.getLocalName().equals(XmlElementNames.ServerDN)) {
606           this.serverDN = reader.readElementValue();
607         } else if (reader.getLocalName().equals(XmlElementNames.ServerVersion)) {
608           reader.readElementValue();
609         } else if (reader.getLocalName().equals(XmlElementNames.AD)) {
610           this.activeDirectoryServer = reader.readElementValue();
611         } else if (reader.getLocalName().equals(XmlElementNames.MdbDN)) {
612           this.mailboxDN = reader.readElementValue();
613         } else if (reader.getLocalName().equals(XmlElementNames.EWSUrl)) {
614           this.exchangeWebServicesUrl = reader.readElementValue();
615         } else if (reader.getLocalName().equals(XmlElementNames.ASUrl)) {
616           this.availabilityServiceUrl = reader.readElementValue();
617         } else if (reader.getLocalName().equals(XmlElementNames.OOFUrl)) {
618           reader.readElementValue();
619         } else if (reader.getLocalName().equals(XmlElementNames.UMUrl)) {
620           this.unifiedMessagingUrl = reader.readElementValue();
621         } else if (reader.getLocalName().equals(XmlElementNames.OABUrl)) {
622           this.offlineAddressBookUrl = reader.readElementValue();
623         } else if (reader.getLocalName().equals(
624             XmlElementNames.PublicFolderServer)) {
625           this.publicFolderServer = reader.readElementValue();
626         } else if (reader.getLocalName().equals(
627             XmlElementNames.Internal)) {
628           OutlookProtocol.loadWebClientUrlsFromXml(reader,
629               this.internalOutlookWebAccessUrls, reader.getLocalName());
630         } else if (reader.getLocalName().equals(
631             XmlElementNames.External)) {
632           OutlookProtocol.loadWebClientUrlsFromXml(reader,
633               this.externalOutlookWebAccessUrls, reader.getLocalName());
634         } else if (reader.getLocalName().equals(
635             XmlElementNames.Ssl)) {
636           String sslStr = reader.readElementValue();
637           this.sslEnabled = sslStr.equalsIgnoreCase("On");
638         } else if (reader.getLocalName().equals(
639             XmlElementNames.SharingUrl)) {
640           this.sharingEnabled = reader.
641               readElementValue().length() > 0;
642         } else if (reader.getLocalName().equals(
643             XmlElementNames.EcpUrl)) {
644           this.ecpUrl = reader.readElementValue();
645         } else if (reader.getLocalName().equals(
646             XmlElementNames.EcpUrl_um)) {
647           this.ecpUrlUm = reader.readElementValue();
648         } else if (reader.getLocalName().equals(
649             XmlElementNames.EcpUrl_aggr)) {
650           this.ecpUrlAggr = reader.readElementValue();
651         } else if (reader.getLocalName().equals(
652             XmlElementNames.EcpUrl_sms)) {
653           this.ecpUrlSms = reader.readElementValue();
654         } else if (reader.getLocalName().equals(
655             XmlElementNames.EcpUrl_mt)) {
656           this.ecpUrlMt = reader.readElementValue();
657         } else if (reader.getLocalName().equals(
658             XmlElementNames.EcpUrl_ret)) {
659           this.ecpUrlRet = reader.readElementValue();
660         } else if (reader.getLocalName().equals(
661             XmlElementNames.EcpUrl_publish)) {
662           this.ecpUrlPublish = reader.readElementValue();
663         } else if (reader.getLocalName().equals(
664             XmlElementNames.ExchangeRpcUrl)) {
665             this.exchangeRpcUrl = reader.readElementValue();
666         } else if (reader.getLocalName().equals(
667             XmlElementNames.GroupingInformation)) {
668             this.groupingInformation = reader.readElementValue();
669         } else {
670           reader.skipCurrentElement();
671         }
672       }
673     } while (!reader.isEndElement(XmlNamespace.NotSpecified,
674         XmlElementNames.Protocol));
675   }

683   private static OutlookProtocolType protocolNameToType(String

695   /**
696    * Loads web client urls from XML.
697    *
698    * @param reader        The reader.
699    * @param webClientUrls The web client urls.
700    * @param elementName   Name of the element.
701    * @throws Exception
702    */
703   private static void loadWebClientUrlsFromXml(EwsXmlReader reader,
704       WebClientUrlCollection webClientUrls, String elementName) throws Exception {
705     do {
706       reader.read();
707 
708       if (reader.getNodeType().getNodeType() == XmlNodeType.START_ELEMENT) {
709         if (reader.getLocalName().equals(XmlElementNames.OWAUrl)) {
710           String authMethod = reader.readAttributeValue(
711               XmlAttributeNames.AuthenticationMethod);
712           String owaUrl = reader.readElementValue();
713           WebClientUrl webClientUrl =
714               new WebClientUrl(authMethod, owaUrl);
715           webClientUrls.getUrls().add(webClientUrl);
716         } else {
717           reader.skipCurrentElement();
718         }
719       }
720     }
721     while (!reader.isEndElement(XmlNamespace.NotSpecified, elementName));
722   }

736   /**
737    * Convert OutlookProtocol to GetUserSettings response.
738    *
739    * @param requestedSettings The requested settings.
740    * @param response          The response.
741    */
742   protected void convertToUserSettings(
743       List<UserSettingName> requestedSettings,
744       GetUserSettingsResponse response) {
745     if (this.getConverterDictionary() != null) {
746       // In English: collect converters that are contained in the requested settings.
747       Map<UserSettingName, IFunc<OutlookProtocol, Object>> converterQuery =
748           new HashMap<UserSettingName, IFunc<OutlookProtocol, Object>>();
749       Map<UserSettingName, IFunc<OutlookProtocol, Object>> t =
750           this.getConverterDictionary();
751       for (Entry<UserSettingName, IFunc<OutlookProtocol, Object>> map : t.entrySet()) {
752         if (requestedSettings.contains(map.getKey())) {
753           converterQuery.put(map.getKey(), map.getValue());
754         }
755       }
756 
757       for (Entry<UserSettingName, IFunc<OutlookProtocol, Object>> kv : converterQuery.entrySet()) {
758         Object value = kv.getValue().func(this);
759         if (value != null) {
760           response.getSettings().put(kv.getKey(), value);
761         }
762       }
763     }
764   }
