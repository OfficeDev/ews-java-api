24  package microsoft.exchange.webservices.data.misc;
25  
26  import microsoft.exchange.webservices.data.core.EwsServiceXmlReader;
27  import microsoft.exchange.webservices.data.core.XmlAttributeNames;
28  import microsoft.exchange.webservices.data.core.XmlElementNames;
29  import microsoft.exchange.webservices.data.core.enumeration.misc.XmlNamespace;
30  import microsoft.exchange.webservices.data.property.complex.EmailAddress;
31  
32  import java.util.ArrayList;
33  import java.util.Collection;
34  import java.util.Iterator;
35  
36  /**
37   * Represents the results of an ExpandGroup operation.
38   */
39  public final class ExpandGroupResults implements Iterable<EmailAddress> {
40  
41    /**
42     * True, if all members are returned. EWS always returns true on ExpandDL,
43     * i.e. all members are returned.
44     */
45    private boolean includesAllMembers;
46  
47    /**
48     * DL members.
49     */
50    private Collection<EmailAddress> members = new ArrayList<EmailAddress>();
51  
52    /**
53     * Initializes a new instance of the class.
54     */
55    public ExpandGroupResults() {
56    }
57  
58    /**
59     * Gets the number of members that were returned by the ExpandGroup
60     * operation. Count might be less than the total number of members in the
61     * group, in which case the value of the IncludesAllMembers is false.
62     *
63     * @return the count
64     */
65    public int getCount() {
66      return this.getMembers().size();
67    }
68  
69    /**
70     * Gets a value indicating whether all the members of the group have been
71     * returned by ExpandGroup.
72     *
73     * @return the includes all members
74     */
75    public boolean getIncludesAllMembers() {
76      return this.includesAllMembers;
77    }
78  
79    /**
80     * Gets the members of the expanded group.
81     *
82     * @return the members
83     */
84    public Collection<EmailAddress> getMembers() {
85      return this.members;
86    }
87  
88    /**
89     * Gets the members of the expanded group.
90     *
91     * @param reader the reader
92     * @throws Exception the exception
93     */
94    public void loadFromXml(EwsServiceXmlReader reader) throws Exception {
95      reader.readStartElement(XmlNamespace.Messages,
96          XmlElementNames.DLExpansion);
97      if (!reader.isEmptyElement()) {
98        int totalItemsInView = reader.readAttributeValue(Integer.class,
99            XmlAttributeNames.TotalItemsInView);
100       this.includesAllMembers = reader.readAttributeValue(Boolean.class,
101           XmlAttributeNames.IncludesLastItemInRange);
102 
103       for (int i = 0; i < totalItemsInView; i++) {
104         EmailAddress emailAddress = new EmailAddress();
105 
106         reader.readStartElement(XmlNamespace.Types,
107             XmlElementNames.Mailbox);
108         emailAddress.loadFromXml(reader, XmlElementNames.Mailbox);
109 
110         this.getMembers().add(emailAddress);
111       }
112 
113       reader.readEndElement(XmlNamespace.Messages,
114           XmlElementNames.DLExpansion);
115     } else {
116       reader.read();
117     }
118   }
119 
120   /**
121    * Returns an iterator over a set of elements of type T.
122    *
123    * @return an Iterator.
124    */
125   @Override
126   public Iterator<EmailAddress> iterator() {
127 
128     return members.iterator();
129   }
130 }
