48  /**
49   * Represents the response to a ExecuteDiagnosticMethod operation
50   */
51  public final class ExecuteDiagnosticMethodResponse extends ServiceResponse {
52  
53  
54    /**
55     * Initializes a new instance of the ExecuteDiagnosticMethodResponse class.
56     *
57     * @param service The service
58     */
59    public ExecuteDiagnosticMethodResponse(ExchangeService service) {
60      super();
61      EwsUtilities.ewsAssert(service != null, "ExecuteDiagnosticMethodResponse.ctor", "service is null");
62    }
63  
64    /**
65     * Reads response elements from XML.
66     *
67     * @throws Exception
68     */
69    @Override
70    protected void readElementsFromXml(EwsServiceXmlReader reader)
71        throws Exception {
72      reader.readStartElement(XmlNamespace.Messages,
73          XmlElementNames.ReturnValue);
74  
75      XMLEventReader returnValueReader = reader.getXmlReaderForNode();
76      //this.returnValue = (Document) new SafeXmlDocument();
77      {
78        this.returnValue = retriveDocument(returnValueReader);
79      }
80  
81      reader.skipCurrentElement();
82      reader.readEndElementIfNecessary(XmlNamespace.Messages,
83          XmlElementNames.ReturnValue);
84    }
85  
86  
87    /**
88     * @return document
89     * @throws javax.xml.parsers.ParserConfigurationException
90     */
91    public Document retriveDocument(XMLEventReader xmlEventReader)
92        throws ParserConfigurationException {
93      DocumentBuilderFactory dbfInstance = DocumentBuilderFactory
94          .newInstance();
95      DocumentBuilder documentBuilder = dbfInstance.newDocumentBuilder();
96      Document document = documentBuilder.newDocument();
97  
98      Element currentElement = document.getDocumentElement();
99  
100     while (xmlEventReader.hasNext()) {
101       XMLEvent xmleve = (XMLEvent) xmlEventReader.next();
102 
103       if (xmleve.getEventType() == XmlNodeType.END_ELEMENT) {
104         Node node = currentElement.getParentNode();
105         if (node instanceof Document) {
106           currentElement = ((Document) node).getDocumentElement();
107         } else {
108           currentElement = (Element) currentElement.getParentNode();
109         }
110       }
111 
112       if (xmleve.getEventType() == XmlNodeType.START_ELEMENT) {
113         // startElement((StartElement) xmleve,doc);
114         StartElement ele = (StartElement) xmleve;
115         Element element = null;
116         element = document.createElementNS(ele.getName()
117             .getNamespaceURI(), ele.getName().getLocalPart());
118 
119         Iterator<Attribute> ite = ele.getAttributes();
120 
121         while (ite.hasNext()) {
122           Attribute attr = ite.next();
123           element.setAttribute(attr.getName().getLocalPart(),
124               attr.getValue());
125         }
126 
127         String xmlns = EwsUtilities.WSTrustFebruary2005Namespace;//"http://schemas.xmlsoap.org/wsdl/";
128         ite = ele.getNamespaces();
129         while (ite.hasNext()) {
130           Namespace ns = (Namespace) ite.next();
131           String name = ns.getPrefix();
132           if (!name.isEmpty()) {
133             element.setAttributeNS(xmlns, name,
134                 ns.getNamespaceURI());
135           } else {
136             xmlns = ns.getNamespaceURI();
137           }
138         }
139 
140         if (currentElement == null) {
141           document.appendChild(element);
142         } else {
143           currentElement.appendChild(element);
144         }
145 
146         currentElement = element;
147         element.setUserData("location", ele.getLocation(), null);
148       }
149     }
150     return document;
151   }
152 
153   private Document returnValue;
154 
155   /**
156    * Gets the return value.
157    */
158   public Document getReturnValue() {
159     return returnValue;
160   }
161 
162   /**
163    * Sets the return value.
164    */
165   private void setReturnValue(Document value) {
166     returnValue = value;
167   }
