45  /**
46   * The Class SubscribeRequest.
47   *
48   * @param <TSubscription> the generic type
49   */
50  abstract class SubscribeRequest<TSubscription extends SubscriptionBase> extends
51      MultiResponseServiceRequest<SubscribeResponse<TSubscription>> {
52  
53    /**
54     * The folder ids.
55     */
56    private FolderIdWrapperList folderIds = new FolderIdWrapperList();
57  
58    /**
59     * The event types.
60     */
61    private List<EventType> eventTypes = new ArrayList<EventType>();
62  
63    /**
64     * The watermark.
65     */
66    private String watermark;
67  
68    /**
69     * Validate request.
70     *
71     * @throws Exception the exception
72     */
73    @Override
74    protected void validate() throws Exception {
75      super.validate();
76      EwsUtilities.validateParam(this.getFolderIds(), "FolderIds");
77      EwsUtilities.validateParamCollection(this.getEventTypes().iterator(),
78          "EventTypes");
79      this.getFolderIds().validate(
80          this.getService().getRequestedServerVersion());
81      // Check that caller isn't trying
82      //to subscribe to Status events.
83      if (this.getEventTypes().contains(EventType.Status)) {
84        throw new ServiceValidationException("Status events can't be subscribed to.");
85      }
86  
87      // If Watermark was specified, make sure it's not a blank string.
88      if (!(this.getWatermark() == null ||
89          this.getWatermark().isEmpty())) {
90        EwsUtilities.validateNonBlankStringParam(this.
91            getWatermark(), "Watermark");
92      }
93  
94      for (EventType eventType : this.getEventTypes()) {
95        EwsUtilities.validateEnumVersionValue(eventType,
96            this.getService().getRequestedServerVersion());
97      }
98  
99    }
100

108   /**
109    * Gets the expected response message count.
110    *

115     return 1;
116   }

118   /**
119    * Gets the name of the XML element.
120    *
121    * @return XML element name
122    */
123   @Override public String getXmlElementName() {
124     return XmlElementNames.Subscribe;
125   }
126 
127   /**
128    * Gets the name of the response XML element.
129    *
130    * @return XML element name
131    */
132   @Override
133   protected String getResponseXmlElementName() {
134     return XmlElementNames.SubscribeResponse;
135   }
136 
137   /**
138    * Gets the name of the response message XML element.
139    *
140    * @return XML element name
141    */
142   @Override
143   protected String getResponseMessageXmlElementName() {
144     return XmlElementNames.SubscribeResponseMessage;
145   }
146 
147   /**
148    * Internal method to write XML elements.
149    *
150    * @param writer the writer
151    * @throws XMLStreamException the XML stream exception
152    * @throws ServiceXmlSerializationException the service xml serialization exception
153    */
154   protected abstract void internalWriteElementsToXml(
155       EwsServiceXmlWriter writer) throws XMLStreamException, ServiceXmlSerializationException;
156 
157   /**
158    * Writes XML elements.
159    *
160    * @param writer the writer
161    * @throws Exception the exception
162    */
163   @Override
164   protected void writeElementsToXml(EwsServiceXmlWriter writer)
165       throws Exception {
166     writer.writeStartElement(XmlNamespace.Messages, this
167         .getSubscriptionXmlElementName());
168 
169     if (this.getFolderIds().getCount() == 0) {
170       writer.writeAttributeValue(XmlAttributeNames.SubscribeToAllFolders,
171           true);
172     }
173 
174     this.getFolderIds().writeToXml(writer, XmlNamespace.Types,
175         XmlElementNames.FolderIds);
176 
177     writer
178         .writeStartElement(XmlNamespace.Types,
179             XmlElementNames.EventTypes);
180     for (EventType eventType : this.getEventTypes()) {
181       writer.writeElementValue(XmlNamespace.Types,
182           XmlElementNames.EventType, eventType);
183     }
184     writer.writeEndElement();
185 
186     if (!(this.getWatermark() == null || this.getWatermark().isEmpty())) {
187       writer.writeElementValue(XmlNamespace.Types,
188           XmlElementNames.Watermark, this.getWatermark());
189     }
190 
191     this.internalWriteElementsToXml(writer);
192 
193     writer.writeEndElement();
194   }
195 
196   /**
197    * Instantiates a new subscribe request.
198    *
199    * @param service the service
200    * @throws Exception
201    */
202   protected SubscribeRequest(ExchangeService service)
203       throws Exception {
204     super(service, ServiceErrorHandling.ThrowOnError);
205     this.setFolderIds(new FolderIdWrapperList());
206     this.setEventTypes(new ArrayList<EventType>());
207   }
