024package microsoft.exchange.webservices.data.property.complex.recurrence.range;
025
026import microsoft.exchange.webservices.data.core.XmlElementNames;
027import microsoft.exchange.webservices.data.property.complex.recurrence.pattern.Recurrence;
028
029import java.util.Date;
030
031/**
032 * Represents recurrence range with no end date.
033 */
034public final class NoEndRecurrenceRange extends RecurrenceRange {
035
036  /**
037   * Initializes a new instance.
038   */
039  public NoEndRecurrenceRange() {
040    super();
041  }
042
043  /**
044   * Initializes a new instance.
045   *
046   * @param startDate the start date
047   */
048  public NoEndRecurrenceRange(Date startDate) {
049    super(startDate);
050  }
051
052  /**
053   * Gets the name of the XML element.
054   *
055   * @return The name of the XML element
056   */
057  public String getXmlElementName() {
058    return XmlElementNames.NoEndRecurrence;
059  }
060
061  /**
062   * Setups the recurrence.
063   *
064   * @param recurrence the new up recurrence
065   * @throws Exception the exception
066   */
067  public void setupRecurrence(Recurrence recurrence) throws Exception {
068    super.setupRecurrence(recurrence);
069
070    recurrence.neverEnds();
071  }
072
073}
