024package microsoft.exchange.webservices.data.core.request;
025
026import microsoft.exchange.webservices.data.core.ExchangeService;
027import microsoft.exchange.webservices.data.core.XmlElementNames;
028import microsoft.exchange.webservices.data.core.enumeration.service.error.ServiceErrorHandling;
029import microsoft.exchange.webservices.data.core.response.MoveCopyFolderResponse;
030import microsoft.exchange.webservices.data.core.enumeration.misc.ExchangeVersion;
031
032/**
033 * Represents a CopyFolder request.
034 */
035public class CopyFolderRequest extends MoveCopyFolderRequest<MoveCopyFolderResponse> {
036
037  /**
038   * Initializes a new instance of the CopyFolderRequest class.
039   *
040   * @param service           the service
041   * @param errorHandlingMode the error handling mode
042   * @throws Exception
043   */
044  public CopyFolderRequest(ExchangeService service, ServiceErrorHandling errorHandlingMode)
045      throws Exception {
046    super(service, errorHandlingMode);
047  }
048
049  /**
050   * Creates the service response.
051   *
052   * @param service       The Service
053   * @param responseIndex Index of the response.
054   * @return Service response.
055   */
056  @Override
057  protected MoveCopyFolderResponse createServiceResponse(
058      ExchangeService service, int responseIndex) {
059    return new MoveCopyFolderResponse();
060  }
061
062  /**
063   * Gets the name of the XML element.
064   *
065   * @return XML element name
066   */
067  @Override public String getXmlElementName() {
068    return XmlElementNames.CopyFolder;
069  }
070
071  /**
072   * Gets the name of the response XML element.
073   *
074   * @return XML element name
075   */
076  @Override
077  protected String getResponseXmlElementName() {
078    return XmlElementNames.CopyFolderResponse;
079  }
080
081  /**
082   * Gets the name of the response message XML element.
083   *
084   * @return XML element name.
085   */
086  @Override
087  protected String getResponseMessageXmlElementName() {
088    return XmlElementNames.CopyFolderResponseMessage;
089  }
090
091  /**
092   * Gets the request version.
093   *
094   * @return Earliest Exchange version in which this request is supported.
095   */
096  @Override
097  protected ExchangeVersion getMinimumRequiredServerVersion() {
098    return ExchangeVersion.Exchange2007_SP1;
099  }
100
101}
