024package microsoft.exchange.webservices.data.core.response;
025
026import microsoft.exchange.webservices.data.core.EwsServiceXmlReader;
027import microsoft.exchange.webservices.data.core.XmlElementNames;
028import microsoft.exchange.webservices.data.core.enumeration.misc.XmlNamespace;
029import microsoft.exchange.webservices.data.property.complex.DelegateUser;
030
031/**
032 * Represents the response to an individual delegate user manipulation (add,
033 * remove, update) operation.
034 */
035public final class DelegateUserResponse extends ServiceResponse {
036
037  /**
038   * The read delegate user.
039   */
040  private boolean readDelegateUser;
041
042  /**
043   * The delegate user.
044   */
045  private DelegateUser delegateUser;
046
047  /**
048   * Initializes a new instance of the class.
049   *
050   * @param readDelegateUser the read delegate user
051   * @param delegateUser     the delegate user
052   */
053  protected DelegateUserResponse(boolean readDelegateUser,
054      DelegateUser delegateUser) {
055    super();
056    this.readDelegateUser = readDelegateUser;
057    this.delegateUser = delegateUser;
058  }
059
060  /**
061   * Reads response elements from XML.
062   *
063   * @param reader the reader
064   * @throws Exception the exception
065   */
066  protected void readElementsFromXml(EwsServiceXmlReader reader)
067      throws Exception {
068    if (this.readDelegateUser) {
069      if (this.delegateUser == null) {
070        this.delegateUser = new DelegateUser();
071      }
072
073      reader.readStartElement(XmlNamespace.Messages,
074          XmlElementNames.DelegateUser);
075
076      this.delegateUser.loadFromXml(reader, XmlNamespace.Messages, reader
077          .getLocalName());
078    }
079  }
080
081  /**
082   * The delegate user that was involved in the operation.
083   *
084   * @return the delegate user
085   */
086  public DelegateUser getDelegateUser() {
087    return this.delegateUser;
088  }
089
090}
