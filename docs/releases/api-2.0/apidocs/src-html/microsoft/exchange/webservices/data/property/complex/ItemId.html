024package microsoft.exchange.webservices.data.property.complex;
025
026import microsoft.exchange.webservices.data.core.XmlElementNames;
027
028/**
029 * Represents the Id of an Exchange item.
030 */
031public class ItemId extends ServiceId {
032
033  /**
034   * Initializes a new instance.
035   */
036  public ItemId() {
037    super();
038  }
039
040  /**
041   * Defines an implicit conversion between string and ItemId.
042   *
043   * @param uniqueId The unique Id to convert to ItemId.
044   * @return An ItemId initialized with the specified unique Id.
045   * @throws Exception the exception
046   */
047  public static ItemId getItemIdFromString(String uniqueId) throws Exception {
048    return new ItemId(uniqueId);
049  }
050
051  /**
052   * Initializes a new instance of ItemId.
053   *
054   * @param uniqueId The unique Id used to initialize the ItemId.
055   * @throws Exception the exception
056   */
057  public ItemId(String uniqueId) throws Exception {
058    super(uniqueId);
059  }
060
061  /**
062   * Gets the name of the XML element.
063   *
064   * @return XML element name.
065   */
066  @Override
067  public String getXmlElementName() {
068    return XmlElementNames.ItemId;
069  }
070}
