024package microsoft.exchange.webservices.data.property.complex;
025
026import microsoft.exchange.webservices.data.attribute.EditorBrowsable;
027import microsoft.exchange.webservices.data.core.enumeration.attribute.EditorBrowsableState;
028import microsoft.exchange.webservices.data.core.enumeration.property.PhoneNumberKey;
029import microsoft.exchange.webservices.data.misc.OutParam;
030
031/**
032 * Represents a dictionary of phone numbers.
033 */
034@EditorBrowsable(state = EditorBrowsableState.Never)
035public final class PhoneNumberDictionary extends DictionaryProperty<PhoneNumberKey, PhoneNumberEntry> {
036
037  /**
038   * Gets the field URI.
039   *
040   * @return Field URI.
041   */
042  @Override
043  protected String getFieldURI() {
044    return "contacts:PhoneNumber";
045  }
046
047  /**
048   * Creates instance of dictionary entry.
049   *
050   * @return New instance.
051   */
052  @Override
053  protected PhoneNumberEntry createEntryInstance() {
054    return new PhoneNumberEntry();
055  }
056
057  /**
058   * Gets  the phone number at the specified key.
059   *
060   * @param key The phone number key.
061   * @return The phone number at the specified key if found; otherwise null.
062   */
063  public String getPhoneNumber(PhoneNumberKey key) {
064    PhoneNumberEntry phoneNumberEntry = this.getEntries().get(key);
065    if (phoneNumberEntry == null) {
066      return null;
067    }
068
069    return phoneNumberEntry.getPhoneNumber();
070  }
071
072  /**
073   * Sets the phone number.
074   *
075   * @param key   the key
076   * @param value the value
077   */
078  public void setPhoneNumber(PhoneNumberKey key, String value) {
079    if (value == null) {
080      this.internalRemove(key);
081    } else {
082      PhoneNumberEntry entry;
083
084      if (this.getEntries().containsKey(key)) {
085        entry = this.getEntries().get(key);
086        entry.setPhoneNumber(value);
087        complexPropertyChanged(entry);
088        this.changed();
089      } else {
090        entry = new PhoneNumberEntry(key, value);
091        this.internalAdd(entry);
092      }
093    }
094  }
095
096  /**
097   * Tries to get the phone number associated with the specified key.
098   *
099   * @param key      the key
100   * @param outparam the outparam
101   * @return true if the Dictionary contains a phone number associated with
102   * the specified key; otherwise, false.
103   */
104  public boolean tryGetValue(PhoneNumberKey key, OutParam<String> outparam) {
105    String phoneNumber = this.getPhoneNumber(key);
106    if (phoneNumber == null) {
107      return false;
108    }
109
110    outparam.setParam(phoneNumber);
111    return true;
112  }
113}
