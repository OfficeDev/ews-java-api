039/**
040 * Testclass for methods of UserConfigurationDictionary
041 */
042@RunWith(JUnit4.class)
043public class UserConfigurationDictionaryTest extends BaseTest {
044
045  /**
046   * Mock for the UserConfigurationDictionary
047   */
048  protected UserConfigurationDictionary userConfigurationDictionary;
049
050  @Before
051  public void setup() throws Exception {
052    // Initialise a UserConfigurationDictionary Testobject
053    this.userConfigurationDictionary = new UserConfigurationDictionary();
054  }
055
056  /**
057   * Adding a Double Value to the Dictionary witch is not allowed
058   *
059   * @throws Exception
060   */
061  @Test(expected = ServiceLocalException.class)
062  public void testAddUnsupportedElementsToDictionary() throws Exception {
063    this.userConfigurationDictionary.addElement("someDouble", 1.0);
064  }
065
066  /**
067   * testAddSupportedElementsToDictionary
068   *
069   * @throws Exception
070   */
071  @Test
072  public void testAddSupportedElementsToDictionary() throws Exception {
073    fillDictionaryWithValidEntries();
074  }
075
076  /**
077   * Fills the Dictionary with
078   *
079   * @throws Exception
080   */
081  private void fillDictionaryWithValidEntries() throws Exception {
082    // Adding Test Values to the Object
083    final int testInt = 1;
084    final long testLong = 1l;
085    final String testString = "someVal";
086    final String[] testStringArray = new String[] {"test1", "test2", "test3"};
087    final Date testDate = new Date();
088    final boolean testBoolean = true;
089    final byte testByte = Byte.decode("0x10");
090    final byte[] testByteArray = testString.getBytes();
091    final Byte[] testByteArray2 = new Byte[testByteArray.length];
092    for (int currentIndex = 0; currentIndex < testByteArray.length; currentIndex++) {
093      testByteArray2[currentIndex] = testByteArray[currentIndex];
094    }
095
096    Assert.assertNotNull(this.userConfigurationDictionary);
097
098    this.userConfigurationDictionary.addElement("someString", testString);
099    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someString"));
100    Assert.assertEquals(testString, this.userConfigurationDictionary.getElements("someString"));
101    Assert.assertTrue(this.userConfigurationDictionary.getElements("someString") instanceof String);
102
103    this.userConfigurationDictionary.addElement("someLong", testLong);
104    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someLong"));
105    Assert.assertEquals(testLong, this.userConfigurationDictionary.getElements("someLong"));
106    Assert.assertTrue(this.userConfigurationDictionary.getElements("someLong") instanceof Long);
107
108    this.userConfigurationDictionary.addElement("someInteger", testInt);
109    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someInteger"));
110    Assert.assertEquals(testInt, this.userConfigurationDictionary.getElements("someInteger"));
111    Assert.assertTrue(this.userConfigurationDictionary.getElements("someInteger") instanceof Integer);
112
113    this.userConfigurationDictionary.addElement("someString[]", testStringArray);
114    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someString[]"));
115    Assert.assertEquals(testStringArray, this.userConfigurationDictionary.getElements("someString[]"));
116    Assert.assertTrue(this.userConfigurationDictionary.getElements("someString[]") instanceof String[]);
117
118    this.userConfigurationDictionary.addElement("someDate", testDate);
119    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someDate"));
120    Assert.assertEquals(testDate, this.userConfigurationDictionary.getElements("someDate"));
121    Assert.assertTrue(this.userConfigurationDictionary.getElements("someDate") instanceof Date);
122
123    this.userConfigurationDictionary.addElement("someBoolean", testBoolean);
124    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someBoolean"));
125    Assert.assertEquals(testBoolean, this.userConfigurationDictionary.getElements("someBoolean"));
126    Assert.assertTrue(this.userConfigurationDictionary.getElements("someBoolean") instanceof Boolean);
127
128    this.userConfigurationDictionary.addElement("someByte", testByte);
129    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someByte"));
130    Assert.assertEquals(testByte, this.userConfigurationDictionary.getElements("someByte"));
131    Assert.assertTrue(this.userConfigurationDictionary.getElements("someByte") instanceof Byte);
132
133    this.userConfigurationDictionary.addElement("someByte[]", testByteArray);
134    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someByte[]"));
135    Assert.assertEquals(testByteArray, this.userConfigurationDictionary.getElements("someByte[]"));
136    Assert.assertTrue(this.userConfigurationDictionary.getElements("someByte[]") instanceof byte[]);
137
138    this.userConfigurationDictionary.addElement("someByte2[]", testByteArray2);
139    Assert.assertTrue(this.userConfigurationDictionary.containsKey("someByte2[]"));
140    Assert.assertEquals(testByteArray2, this.userConfigurationDictionary.getElements("someByte2[]"));
141    Assert.assertTrue(this.userConfigurationDictionary.getElements("someByte2[]") instanceof Byte[]);
142  }
143
144  /**
145   * Tests the Method writeElementsToXml(...)
146   * with all valid Elements
147   */
148  @Test
149  public void testWriteElementsToXml() throws Exception {
150    // Mock up needed Classes
151    OutputStream output = new ByteArrayOutputStream();
152    EwsServiceXmlWriter testWriter = new EwsServiceXmlWriter(exchangeServiceBaseMock, output);
153
154    // Adding Test Values to the Object
155    fillDictionaryWithValidEntries();
156
157    // Write the Elements
158    this.userConfigurationDictionary.writeElementsToXml(testWriter);
159  }
