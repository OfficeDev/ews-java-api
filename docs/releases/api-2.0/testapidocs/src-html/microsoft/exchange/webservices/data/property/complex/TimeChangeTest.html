021package microsoft.exchange.webservices.data.property.complex;
022
023import java.util.Calendar;
024import java.util.TimeZone;
025
026import javax.xml.bind.DatatypeConverter;
027
028import microsoft.exchange.webservices.data.core.EwsUtilities;
029import microsoft.exchange.webservices.data.misc.Time;
030
031import org.junit.Assert;
032import org.junit.Test;
033import org.junit.runner.RunWith;
034import org.junit.runners.JUnit4;
035
036@RunWith(JUnit4.class)
037public class TimeChangeTest {
038
039  private static String time = "03:00:00";
040  private static String time_fail1 = "21:32";
041  private static String time_fail2 = "25:25:10";
042  private static String time_fail3 = "-10:00:00";
043
044  private static String dateUTC = "2001-10-27Z";
045  private static String date_fail1 = "2001-10-32";
046  private static String date_fail2 = "2001-13-26+02:00";
047  private static String date_fail3 = "01-10-26";
048
049  @Test
050  public void testDateUTC() {
051    Assert.assertEquals("2001-10-27Z", testDate(dateUTC));
052  }
053
054  private String testDate(String value) {
055    Calendar cal = DatatypeConverter.parseDate(value);
056    cal.setTimeZone(TimeZone.getTimeZone("UTC"));
057    String XSDate = EwsUtilities.dateTimeToXSDate(cal.getTime());
058    return XSDate;
059  }
060
061  @Test(expected = IllegalArgumentException.class)
062  public void testDateFail1() {
063    testDate(date_fail1);
064  }
065
066  @Test(expected = IllegalArgumentException.class)
067  public void testDateFail2() {
068    testDate(date_fail2);
069  }
070
071  @Test(expected = IllegalArgumentException.class)
072  public void testDateFail3() {
073    testDate(date_fail3);
074  }
075
076  private String testTime(String value) {
077    Calendar cal = DatatypeConverter.parseTime(value);
078    Time time = new Time(cal.getTime());
079    return time.toXSTime();
080  }
081
082  @Test(expected = IllegalArgumentException.class)
083  public void testTimeFail1() {
084    testTime(time_fail1);
085  }
086
087  @Test(expected = IllegalArgumentException.class)
088  public void testTimeFail2() {
089    testTime(time_fail2);
090  }
091
092  @Test(expected = IllegalArgumentException.class)
093  public void testTimeFail3() {
094    testTime(time_fail3);
095  }
096
097  @Test
098  public void testTimeValues() {
099    Assert.assertEquals("{0:00}:{1:00}:{2:00},3,0,0", testTime(time));
100  }
101
102}
