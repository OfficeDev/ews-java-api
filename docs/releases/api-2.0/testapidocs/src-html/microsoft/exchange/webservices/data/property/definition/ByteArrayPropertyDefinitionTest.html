024package microsoft.exchange.webservices.data.property.definition;
025
026import static org.junit.Assert.assertEquals;
027import static org.junit.Assert.assertNotNull;
028
029import microsoft.exchange.webservices.data.core.enumeration.misc.ExchangeVersion;
030import microsoft.exchange.webservices.data.core.enumeration.property.PropertyDefinitionFlags;
031import org.apache.commons.codec.binary.Base64;
032import org.junit.Before;
033import org.junit.Test;
034import org.junit.runner.RunWith;
035import org.junit.runners.JUnit4;
036
037import java.util.EnumSet;
038
039@RunWith(JUnit4.class)
040public class ByteArrayPropertyDefinitionTest {
041
042  private ByteArrayPropertyDefinition testObject;
043
044  private static final String TEST_STRING = "Lorem ipsum dolor sit amet";
045  private static final String BASE64_ENCODEDSTRING = Base64.encodeBase64String(TEST_STRING.getBytes());
046
047  /**
048   * setup
049   */
050  @Before
051  public void init(){
052    this.testObject =
053        new ByteArrayPropertyDefinition("myTestObject", "myTestUri",
054                                        EnumSet.of(PropertyDefinitionFlags.None),
055                                        ExchangeVersion.Exchange2010_SP2);
056  }
057
058  /**
059   * Test for ByteArrayPropertyDefinition.toString()
060   * This Test should guarantee that toString() byte encoding works
061   */
062  @Test
063  public void testToString(){
064    String result = testObject.toString(TEST_STRING.getBytes());
065    assertNotNull(result);
066    assertEquals(BASE64_ENCODEDSTRING, result);
067  }
068
069}
