024package microsoft.exchange.webservices.data.core.service.items;
025
026import static org.junit.Assert.assertThat;
027
028import microsoft.exchange.webservices.base.BaseTest;
029import microsoft.exchange.webservices.data.core.service.item.Task;
030import microsoft.exchange.webservices.data.core.exception.service.local.ServiceObjectPropertyException;
031import org.hamcrest.core.IsEqual;
032import org.hamcrest.core.IsInstanceOf;
033import org.hamcrest.core.IsNot;
034import org.hamcrest.core.IsNull;
035import org.junit.Before;
036import org.junit.Test;
037import org.junit.runner.RunWith;
038import org.junit.runners.JUnit4;
039
040/**
041 * Testclass for methods of Task
042 */
043@RunWith(JUnit4.class)
044public class TaskTest extends BaseTest {
045
046  /**
047   * Mock for the Task
048   */
049  private Task taskMock;
050
051  /**
052   * Setup Mocks
053   *
054   * @throws Exception
055   */
056  @Before
057  public void setup() throws Exception {
058    this.taskMock = new Task(TaskTest.exchangeServiceMock);
059  }
060
061  /**
062   * Test for reading the value before it is assigned
063   *
064   * @throws Exception
065   */
066  @Test(expected = ServiceObjectPropertyException.class)
067  public void testInitialValuePercent() throws Exception {
068    taskMock.getPercentComplete();
069  }
070
071  /**
072   * Test for adding 0.0 as percentCompleted
073   *
074   * @throws Exception
075   */
076  @Test
077  public void testAddZeroPercent() throws Exception {
078    final Double targetValue = 0.0;
079    taskMock.setPercentComplete(targetValue);
080
081    assertThat(taskMock.getPercentComplete(), IsNot.not(IsNull.nullValue()));
082    assertThat(taskMock.getPercentComplete(), IsInstanceOf.instanceOf(Double.class));
083    assertThat(taskMock.getPercentComplete(), IsEqual.equalTo(targetValue));
084  }
085
086  /**
087   * Test for adding 100.0 as percentCompleted
088   *
089   * @throws Exception
090   */
091  @Test
092  public void testAdd100Percent() throws Exception {
093    final Double targetValue = 100.0;
094    taskMock.setPercentComplete(targetValue);
095
096    assertThat(taskMock.getPercentComplete(), IsNot.not(IsNull.nullValue()));
097    assertThat(taskMock.getPercentComplete(), IsInstanceOf.instanceOf(Double.class));
098    assertThat(taskMock.getPercentComplete(), IsEqual.equalTo(targetValue));
099  }
100
101  /**
102   * Test for adding Double.MAX_VALUE as percentCompleted
103   *
104   * @throws Exception
105   */
106  @Test(expected = IllegalArgumentException.class)
107  public void testAddMaxDoublePercent() throws Exception {
108    final Double targetValue = Double.MAX_VALUE;
109    taskMock.setPercentComplete(targetValue);
110  }
111
112  /**
113   * Test for adding -0.1 as percentCompleted
114   *
115   * @throws Exception
116   */
117  @Test(expected = IllegalArgumentException.class)
118  public void testAddInvalidPercent() throws Exception {
119    final Double targetValue = -0.1;
120    taskMock.setPercentComplete(targetValue);
121  }
122
123  /**
124   * Test for adding +100.1 as percentCompleted
125   *
126   * @throws Exception
127   */
128  @Test(expected = IllegalArgumentException.class)
129  public void testAddInvalidPercent2() throws Exception {
130    final Double targetValue = +100.1;
131    taskMock.setPercentComplete(targetValue);
132  }
133
134  /**
135   * Test for adding Double.NaN as percentCompleted
136   *
137   * @throws Exception
138   */
139  @Test(expected = IllegalArgumentException.class)
140  public void testAddNanDoublePercent() throws Exception {
141    final Double targetValue = Double.NaN; // closest Value to Zero
142    taskMock.setPercentComplete(targetValue);
143  }
144
145  /**
146   * Test for checking if the value changes in case of a thrown exception
147   *
148   * @throws Exception
149   */
150  @Test
151  public void testDontChangeValueOnException() throws Exception {
152    // set valid value
153    final Double targetValue = 50.5;
154    taskMock.setPercentComplete(targetValue);
155
156    assertThat(taskMock.getPercentComplete(), IsNot.not(IsNull.nullValue()));
157    assertThat(taskMock.getPercentComplete(), IsInstanceOf.instanceOf(Double.class));
158    assertThat(taskMock.getPercentComplete(), IsEqual.equalTo(targetValue));
159
160    final Double invalidValue = -0.1;
161    try {
162      taskMock.setPercentComplete(invalidValue);
163    } catch (IllegalArgumentException ex) {
164      // ignored
165    }
166
167    assertThat(taskMock.getPercentComplete(), IsNot.not(IsNull.nullValue()));
168    assertThat(taskMock.getPercentComplete(), IsInstanceOf.instanceOf(Double.class));
169    assertThat(taskMock.getPercentComplete(), IsEqual.equalTo(targetValue));
170  }
171}
